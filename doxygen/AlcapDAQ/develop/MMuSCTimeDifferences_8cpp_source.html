174 int FillCoincidenceTimeHistogram(int tMusc,const std::vector<TPulseIsland*>& pulses,
175  int start, TH1 *h, TH2 *hAmp, double t_low, double t_high,
176  bool PP, TH1 *h_PP, TH2 *hAmp_PP, const std::string& bank_name)
177 {
178  for(int p = start; p < (int)pulses.size(); ++p){
179 
180  double tDiff = GetTime(pulses[p],gSetup, bank_name) - tMusc;
181  if(tDiff < t_low){
182  start = p;
183  continue;
184  }
185  if(tDiff > t_high) break;
186  h->Fill(tDiff);
187  hAmp->Fill(tDiff, pulses[p]->GetPulseHeight());
188  if(PP){
189  h_PP->Fill(tDiff);
190  hAmp_PP->Fill(tDiff, pulses[p]->GetPulseHeight());
191  }
192  }
193 
194  return start;
195 }

197 bool IsPileupProtected(double t,const std::vector<TPulseIsland*>& p, double window, int *lPP, const std::string& bank_name){
198  if(t < window || t > 1.12E8-window) return false; // very basic bookending
199  for(unsigned i = *lPP; i < p.size(); ++i){
200  double t2 = GetTime(p[i],gSetup,bank_name);
201  if(t2 < t-window){
202  *lPP = i;
203  }
204  if(t2 > t+window) break;
205  if(t != t2 && t2 > t-window && t2 < t+window) return false;
206  }
207  return true;
208 }
