334 BOOL dt5720_update_digitizer()
335 {
336  CAEN_DGTZ_ErrorCode ret;
337  /* General CAEN settings */
338 
339  switch(S_DT5720_ODB.software_trigger_mode)
340  {
341  case 0:
342  ret = CAEN_DGTZ_SetSWTriggerMode(handle,CAEN_DGTZ_TRGMODE_DISABLED);
343  break;
344  case 1:
345  ret = CAEN_DGTZ_SetSWTriggerMode(handle,CAEN_DGTZ_TRGMODE_ACQ_ONLY);
346  break;
347  case 2:
348  ret = CAEN_DGTZ_SetSWTriggerMode(handle,CAEN_DGTZ_TRGMODE_EXTOUT_ONLY);
349  break;
350  case 3:
351  ret = CAEN_DGTZ_SetSWTriggerMode(handle,CAEN_DGTZ_TRGMODE_ACQ_AND_EXTOUT);
352  break;
353  default:
354  ret = CAEN_DGTZ_SetSWTriggerMode(handle,CAEN_DGTZ_TRGMODE_DISABLED);
355  cm_msg(MINFO,"dt5720_update_digitizer","Invalid software trigger mode. Set to DISABLED. (%i)",(int)S_DT5720_ODB.software_trigger_mode);
356  break;
357  }
358  if(is_caen_error(ret,__LINE__,"dt5720_update_digitizer")) return false;
359 
360  switch(S_DT5720_ODB.hardware_trigger_mode)
361  {
362  case 0:
363  ret = CAEN_DGTZ_SetExtTriggerInputMode(handle,CAEN_DGTZ_TRGMODE_DISABLED);
364  break;
365  case 1:
366  ret = CAEN_DGTZ_SetExtTriggerInputMode(handle,CAEN_DGTZ_TRGMODE_ACQ_ONLY);
367  break;
368  case 2:
369  ret = CAEN_DGTZ_SetExtTriggerInputMode(handle,CAEN_DGTZ_TRGMODE_EXTOUT_ONLY);
370  break;
371  case 3:
372  ret = CAEN_DGTZ_SetExtTriggerInputMode(handle,CAEN_DGTZ_TRGMODE_ACQ_AND_EXTOUT);
373  break;
374  default:
375  ret = CAEN_DGTZ_SetExtTriggerInputMode(handle,CAEN_DGTZ_TRGMODE_DISABLED);
376  cm_msg(MINFO,"dt5720_update_digitizer","Invalid hardware trigger mode. Set to DISABLED.");
377  break;
378  }
379  if(is_caen_error(ret,__LINE__,"dt5720_update_digitizer")) return false;
380 
381  uint32_t caen_self_trigger_mask = 0b1111;
382  switch(S_DT5720_ODB.self_trigger_mode)
383  {
384  case 0:
385  ret = CAEN_DGTZ_SetChannelSelfTrigger(handle,CAEN_DGTZ_TRGMODE_DISABLED,caen_self_trigger_mask);
386  break;
387  case 1:
388  ret = CAEN_DGTZ_SetChannelSelfTrigger(handle,CAEN_DGTZ_TRGMODE_ACQ_ONLY,caen_self_trigger_mask);
389  break;
390  case 2:
391  ret = CAEN_DGTZ_SetChannelSelfTrigger(handle,CAEN_DGTZ_TRGMODE_EXTOUT_ONLY,caen_self_trigger_mask);
392  break;
393  case 3:
394  ret = CAEN_DGTZ_SetChannelSelfTrigger(handle,CAEN_DGTZ_TRGMODE_ACQ_AND_EXTOUT,caen_self_trigger_mask);
395  break;
396  default:
397  ret = CAEN_DGTZ_SetChannelSelfTrigger(handle,CAEN_DGTZ_TRGMODE_DISABLED,caen_self_trigger_mask);
398  cm_msg(MINFO,"dt5720_update_digitizer","Invalid channel self trigger mode. Set to DISABLED. (%i)",(int)S_DT5720_ODB.self_trigger_mode);
399  break;
400  }
401  if(is_caen_error(ret,__LINE__,"dt5720_update_digitizer")) return false;
402 
403  // According to correspondence with CAEN, second argument (channel nuumber) is ignored.
404  // But just in case...
405  if(S_DT5720_ODB.trigger_positive_edge)
406  {
407  ret = CAEN_DGTZ_SetTriggerPolarity(handle,0,CAEN_DGTZ_TriggerOnRisingEdge);
408  ret = CAEN_DGTZ_SetTriggerPolarity(handle,1,CAEN_DGTZ_TriggerOnRisingEdge);
409  ret = CAEN_DGTZ_SetTriggerPolarity(handle,2,CAEN_DGTZ_TriggerOnRisingEdge);

415  ret = CAEN_DGTZ_SetTriggerPolarity(handle,1,CAEN_DGTZ_TriggerOnFallingEdge);
416  ret = CAEN_DGTZ_SetTriggerPolarity(handle,2,CAEN_DGTZ_TriggerOnFallingEdge);
417  ret = CAEN_DGTZ_SetTriggerPolarity(handle,3,CAEN_DGTZ_TriggerOnFallingEdge);
418  }
419  if(is_caen_error(ret,__LINE__,"dt5720_update_digitizer")) return false;
420 
421  if(S_DT5720_ODB.max_events_per_block >= 2 && S_DT5720_ODB.max_events_per_block <= 1024)
422  {
423  ret = CAEN_DGTZ_SetMaxNumEventsBLT(handle,S_DT5720_ODB.max_events_per_block);
424  }
425  else
426  {
427  ret = CAEN_DGTZ_SetMaxNumEventsBLT(handle,1024);

526  ret = CAEN_DGTZ_SetChannelTriggerThreshold(handle,

530  ret = CAEN_DGTZ_SetChannelTriggerThreshold(handle,
531  2,
532  analog2adc_trigger(2));
533  if(is_caen_error(ret,__LINE__,"dt5720_update_digitizer")) return false;
534  ret = CAEN_DGTZ_SetChannelTriggerThreshold(handle,
535  3,
536  analog2adc_trigger(3));
537  if(is_caen_error(ret,__LINE__,"dt5720_update_digitizer")) return false;
538 
539  // =====================================================================================
540  // Record length
541  // =====================================================================================
542  ret = CAEN_DGTZ_SetRecordLength(handle, S_DT5720_ODB.wf_length);

546  return FE_ERR_HW;

551  // ======================================================================================
552  uint32_t data;

554  if(is_caen_error(ret,__LINE__-1,"dt5720_update_digitizer")) return FE_ERR_HW;
555 
556  // Bit 4 must always be set to 1
557  data |= (1<<4);
558  // enable trigger overlap
559  data |= (1<<1);
560  ret = CAEN_DGTZ_WriteRegister(handle, CAEN_DGTZ_BROAD_CH_CTRL_ADD, data);
561  if(is_caen_error(ret,__LINE__-1,"dt5720_update_digitizer")) return FE_ERR_HW;
562 
563  data_buffer_size = 32*1024*1024;

565
