142 INT MDQ_DigitizerOverflow(EVENT_HEADER *pheader, void *pevent)
143 {
144  // Get the event number
145  int midas_event_number = pheader->serial_number;
146 
147  bool pulse_overflow = 0;
148  int overflow_value = 0;

150  // Some typedefs
151  typedef map<string, vector<TPulseIsland*> > TStringPulseIslandMap;
152  typedef pair<string, vector<TPulseIsland*> > TStringPulseIslandPair;
153  typedef map<string, vector<TPulseIsland*> >::iterator map_iterator;
154 
155  // Fetch a reference to the gData structure that stores a map
156  // of (bank_name, vector<TPulseIsland*>) pairs
157  TStringPulseIslandMap& pulse_islands_map =
158  gData->fPulseIslandToChannelMap;
159 
160  // Loop over the map and get each bankname, vector pair
161  for (map_iterator mapIter = pulse_islands_map.begin(); mapIter != pulse_islands_map.end(); ++mapIter)
162  {
163  std::string bankname = mapIter->first;
164  std::string detname = gSetup->GetDetectorName(bankname);
165  std::vector<TPulseIsland*> thePulses = mapIter->second;
166 
167  int n_bits = gSetup->GetNBits(bankname);
168  overflow_value = std::pow(2, n_bits) - 1;
169 
170  // Loop over the TPulseIslands and plot the histogram
171  for (std::vector<TPulseIsland*>::iterator pulseIter = thePulses.begin(); pulseIter != thePulses.end(); ++pulseIter) {
172 
173  pulse_overflow = 0;
174 
175  //loop over samples, check for overflow value
176  const std::vector<int>& theSamples = (*pulseIter)->GetSamples();
177  for (std::vector<int>::const_iterator sampleIter = theSamples.begin(); sampleIter != theSamples.end(); ++sampleIter){
178  int sample_number = sampleIter - theSamples.begin();
179  int sample_value = *sampleIter;
180 
181  if (sample_value >= overflow_value){
182  //printf(" sample overflow! sample #%d value %d midas event #%d bankname %s detname %s\n",sample_number,sample_value,midas_event_number,bankname.c_str(),detname.c_str());
183  pulse_overflow = 1;
184 
185  }
186 
187 
188  std::string binname = bankname + "(" + detname + ")";
189  hDQ_DigitizerOverflow_Total->Fill(binname.c_str(),1);
190  hDQ_DigitizerOverflow_TotalByEvent->Fill(binname.c_str(),midas_event_number,1);
191  if ((pulse_overflow)){
192  hDQ_DigitizerOverflow_Fraction->Fill(binname.c_str(),1);
193  hDQ_DigitizerOverflow_FractionByEvent->Fill(binname.c_str(),midas_event_number,1);
194  //printf(" total pulses %d integral %f\n",total_pulses,hDQ_ADCSampleOverflow->Integral());
195  }

197  }

201  return SUCCESS;
