135 void pulse_islands_stitch(std::vector<TPulseIsland*>& pulses) {
136  unsigned int nPulses = pulses.size();
137  std::vector<int> next_samples, current_samples;
138  TPulseIsland* temp_pulse;
139  for (unsigned int iPulse = 0; iPulse < nPulses - 1; ++iPulse) {
140  next_samples = pulses[iPulse + 1]->GetSamples();
141  // If the next pulse is less than the set number of samples,
142  // it's a continuation of this pulse
143  while (next_samples.size() < nSamples) {
144  current_samples = pulses[iPulse]->GetSamples();
145  for (unsigned int i = 0; i < next_samples.size(); ++i)
146  current_samples.push_back(next_samples[i]);
147  temp_pulse = pulses[iPulse];
148  pulses[iPulse] = new TPulseIsland(temp_pulse->GetTimeStamp(), current_samples, temp_pulse->GetBankName());
149  delete temp_pulse;
150  delete pulses[iPulse + 1];
151  pulses.erase(pulses.begin() + iPulse + 1);
152  if (!(iPulse < --nPulses - 1)) break;
153  next_samples = pulses[iPulse + 1]->GetSamples();
154  }
155  }
