127 Int_t Main_event_loop(TTree* dataTree,ARGUMENTS& arguments){
128 /*************************************************************************\
129 | Loop over tree entries and call histogramming modules. |
130 \*************************************************************************/
131  Long64_t nentries = dataTree->GetEntriesFast();
132  printf("There are %d entries\n",(int)nentries);
133  std::cout<<"Processing file, which may take a while. "
134  "Have patience young padawan.."<<std::endl;
135 
136  //set up the input data
137  if (g_event){
138  g_event->Clear("C");
139  dataTree->SetBranchAddress("Event",&g_event);
140  }
141 
142  // How many entries should we loop over?
143  Long64_t start = 0;
144  Long64_t stop = nentries;
145  if(arguments.stop>0){
146  if(arguments.start>0)
147  start = (Long64_t)(arguments.start-1);
148  if((Long64_t)arguments.stop < nentries)
149  stop = (Long64_t)arguments.stop;
150  }
151  else if((Long64_t)arguments.start < nentries && arguments.start > 0){
152  stop = (Long64_t)arguments.start;
153  }
154  gTotalEntries=&stop;
155  // wind the file on to the first event
156  dataTree->GetEntry(start);
157 
158  // Get the first and last module to run with
159  modules::list::iterator first_module = modules::navigator::Instance()->Begin();
160  modules::list::iterator last_module = modules::navigator::Instance()->End();
161  modules::list::iterator it_mod;
162 
163  int q = 0;
164  for (it_mod=first_module; it_mod != last_module; it_mod++) {
165  q |= it_mod->second->BeforeFirstEntry(g_event, TSetupData::Instance());
166  }
167  if(q) {
168  printf("Error while preprocessing first entry (%d)\n",(Int_t)start);
169  return q;
170  }
171 
172  q = 0;
173  //process entries
174  Long64_t jentry;
175  gEntryNumber=&jentry;
176  for ( jentry=start; jentry<stop;jentry++) {
177  if(g_event){
178  g_event->Clear("C");
179  ClearGlobalData(g_event);
180  dataTree->SetBranchAddress("Event",&g_event);
181  }
182 
183  if (jentry%100 == 0) {
184  printf("Completed %lld events out of %lld\n", jentry, stop);
185  }
186  // Let's get the next event
187  dataTree->GetEntry(jentry);
188 
189  for (it_mod=first_module; it_mod != last_module; it_mod++) {
190  q |= it_mod->second->ProcessGenericEntry(g_event,TSetupData::Instance());
191  //if(q) break;
192  }
193  if(q){
194  printf("q was non-zero when jentry was %lld\n",jentry);
195  break;
196  }
197

199  //gAnalysedPulseMapWrapper->ShowInfo();
200  gAnalysedPulseTree->Fill();
201  }
202

204  q = 0;

206  q |= it_mod->second->AfterLastEntry(g_event,TSetupData::Instance());

211  }
212
