189 INT MDQ_TDCCheck(EVENT_HEADER *pheader, void *pevent)
190 {
191  // Get the event number
192  int midas_event_number = pheader->serial_number;
193 
194  // Some typedefs
195  typedef map<string, vector<TPulseIsland*> > TStringPulseIslandMap;
196  typedef pair<string, vector<TPulseIsland*> > TStringPulseIslandPair;
197  typedef map<string, vector<TPulseIsland*> >::iterator map_iterator;
198 
199  // Fetch a reference to the gData structure that stores a map
200  // of (bank_name, vector<TPulseIsland*>) pairs
201  TStringPulseIslandMap& pulse_islands_map =
202  gData->fPulseIslandToChannelMap;
203 
204  // Get a pointer to the hit data in the TDC
205  // NB copied and modified from MMuSCAnalysisC.cpp
206  channel_hit *hit_bank;
207  int hit_bank_size = bk_locate(pevent, "HITS", (DWORD *) &hit_bank);
208  hit_bank_size = hit_bank_size * sizeof(DWORD) / sizeof(channel_hit);
209 
210  // At the moment just loop through the hits and print the information
211  // Parameter values have been obtained from MMuSCAnalysisMQL.cpp and MMuPC1AnalysisMQL.cpp
212  for (int i = 0; i < hit_bank_size; ++i) {
213  if (hit_bank[i].parameter == 6011) {
214  // printf("muSC hit! Hit #%d: time = %f, parameter = %d\n", i, hit_bank[i].time, hit_bank[i].parameter);
215  hDQ_TDCCheck_muSc->Fill(1);
216  hDQ_TDCCheck_muSc_time->Fill(hit_bank[i].time);
217  hDQ_TDCCheck_muSc_rate->Fill(1);
218 
219  // Plot the time difference between the time as given by the TDC and as given by the BU CAEN
220  std::string detname = "muSc";
221  std::string bankname = gSetup->GetBankName(detname);
222  std::vector<TPulseIsland*> theMuScPulses = pulse_islands_map[bankname];
223 
224  // Loop over the TPulseIslands and plot the histogram
225  for (std::vector<TPulseIsland*>::iterator pulseIter = theMuScPulses.begin(); pulseIter != theMuScPulses.end(); ++pulseIter) {
226 
227  // Get the timestamp of the TPI in ns
228  int time_stamp = (*pulseIter)->GetTimeStamp();
229  double clock_tick_in_ns = (*pulseIter)->GetClockTickInNs();
230  double block_time = time_stamp * clock_tick_in_ns;
231 
232  hDQ_TDCCheck_TDiff->Fill(block_time - hit_bank[i].time);
233  }
234  }
235 
236  else if (hit_bank[i].parameter == 6002)
237  // printf("muSCA hit! Hit #%d: time = %f, parameter = %d\n", i, hit_bank[i].time, hit_bank[i].parameter);
238  hDQ_TDCCheck_muScA->Fill(1);
239 
240  else if (hit_bank[i].parameter >= 4001 && hit_bank[i].parameter <= 4074)
241  // printf("muPC hit! Hit #%d: time = %f, parameter = %d\n", i, hit_bank[i].time, hit_bank[i].parameter);
242  hDQ_TDCCheck_muPC->Fill(1);
243  else
244  // printf("Unknown hit! Hit #%d: time = %f, parameter = %d\n", i, hit_bank[i].time, hit_bank[i].parameter);
245  hDQ_TDCCheck_Unknown->Fill(hit_bank[i].parameter);
246  }
247 
248  return SUCCESS;
249 }
