140  // The value we're looping over is the bank names in the ODB.
141  // Some corresponding detector names are not real,
142  // such as "blank". We hope these histograms do not exist so

144  for (unsigned int i = 0; i < fODB.GetNDets(); ++i) {
145  // If channel is implicitly disabled (named blank or ZZZZ),
146  // simply copy values from file ODB and disable.
147  if (fODB.GetBanks()[i] == "ZZZZ" || fODB.GetDets()[i] == "blank") {
148  fCorrections.Add(fODB, i);
149  fCorrections.Disable();
150  continue;

153  TH1* shapes;
154  TH1* timing;
155  hist_file.GetObject(("DataQuality_LowLevel/hDQ_PulseShapes_ProjectionY_" + fODB.GetDets()[i] + "_" + fODB.GetBanks()[i]).c_str(), shapes);
156  hist_file.GetObject(("DataQuality_LowLevel/hDQ_muScTDiff_" + fODB.GetDets()[i] + "_" + fODB.GetBanks()[i]).c_str(), timing);
157 
158  // Only if both histograms are present and filled are corrections estimated
159  // Mark all other channels as disabled.
160  // Detector muSc is the exception, which is expected to have an empty
161  // timing correlation histogram.
162  if (!shapes) {
163  fCorrections.Add(fODB, i);
164  std::cout <<
165  "ODBCheck WARNING: Shapes histogram not found! Corrections not included for " <<
166  fODB.GetDets()[i] << "_" << fODB.GetBanks()[i] << "..." <<
167  std::endl;
168  fCorrections.Disable();
169  } else if (!timing) {
170  fCorrections.Add(fODB, i);
171  std::cout <<

174  std::endl;
175  fCorrections.Disable();
176  } else if (!shapes->GetEntries()) {

178  std::cout <<
